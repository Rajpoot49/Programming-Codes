Program to Concatenate Two Strings

#include <iostream>
using namespace std;

int main()
{
    string s1, s2, result;

    cout << "Enter first string: ";
    cin>>s1;

    cout << "Enter second string: ";
    cin>>s2;

    result = s1 + s2;

    cout << "After concatenation: "<< result;

    return 0;
}
