Program to display string in center of screen

.MODEL SMALL
.STACK 100H
PRNSTR MACRO MSG
    MOV AH,09H
    LEA DX,MSG
    INT 21H
ENDM
DATA SEGMENT
    MSG1 DB "ASSEMBLY LANGUAGE $"
DATA ENDS

CODE SEGMENT
START:    
    MOV AX,DATA
    MOV DS,AX
    MOV AX,0600H
    MOV BH,71H
    MOV CX,000H    
    MOV DX,184H    
    INT 10H
    MOV AH,02H
    MOV BH,00H
    MOV DH,0CH
    MOV DL,23H
    INT 10H
    PRNSTR MSG1
    MOV AH,4CH
    INT 21H
CODE ENDS
END START
